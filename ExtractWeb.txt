f1=open("poon.txt","r")  #removing extras of trip
n1=open("poontrip.txt","w+")
for line in f1:
    rep=re.sub("<.*","",line)
    match=re.sub("[.,/<;!]|(but)",". \n",rep)
    n1.write(match)
f1.close()
n1.close()
